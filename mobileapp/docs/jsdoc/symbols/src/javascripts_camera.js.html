 28 /**
 29 	Calling native interface for make photo
 30 	@param {integer} type Type of photo (1 - Box photo, 2 - Document photo).
 31 */
 32 function make_photo(type) {
 33 	if(type == 1) {
 34 	  navigator.camera.getPicture( boxSuccess, cameraError, {
 35 		quality: 30,
 36 		targetWidth:640,
 37 		targetHeight:480,
 38 		destinationType: navigator.camera.DestinationType.DATA_URL,
 39 		encodingType: navigator.camera.EncodingType.JPEG
 40       });
 41 	}
 42 	if(type == 2) {
 43 	  navigator.camera.getPicture( podSuccess, cameraError, {
 44 		targetWidth:1024,
 45 		targetHeight:768,
 46 		quality: 30,
 47 		destinationType: navigator.camera.DestinationType.FILE_URI,
 48 		encodingType: navigator.camera.EncodingType.JPEG
 49       });
 50 	}
 51 }
 52 /**
 53 	Calling scan barcode plugin
 54 */
 55 function scanbarcode()
 56 {
 57 	if(scannerrun) return;
 58 	scannerrun = true;
 59 	window.plugins.barcodeScanner.scan(
 60 		function(result) {
 61 			if (result.cancelled)
 62 				console.log("the user cancelled the scan");
 63 			else
 64 				$('#id-shipment-value').val(result.text);
 65 			scannerrun = false;
 66 		},
 67 		function(error) {
 68 			alert(error);
 69 			scannerrun = false;
 70 		}
 71 	);
 72 }
