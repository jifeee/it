 65 /**
 66    Send request to server
 67    @param {string} url Url address
 68    @param {string} type Type of request (get,post)
 69    @param {array} data List of parameters
 70    @param {function} success (optional) Success callback
 71    @param {function} fail (optional) Fail callback
 72    @param {bool} async (optional) asynchronous flag
 73 */

 75 function send_request(url, type, data, success, fail, async)
 76 {
 77 	if(async == null) async = true;
 78     $.ajax({
 79         url: url,
 80         type: type,
 81         data: data,
 82 		async: async,
 83         timeout: 5000
 84     }).done(function(data){
 85         if(typeof(success) == 'function') success(data);
 86     }).fail(function(error){
 87 		if(error.status == 401)
 88 			alert(t('error_not_auth'));
 89 		else 
 90 			if(check_connection()) if(typeof(fail) == 'function') fail(error);
 91     }).complete(function(jqXHR, textStatus){
 92         var response = JSON.stringify(jqXHR);
 93     });
 94 }

651 /**
652 	Check email address
653 	@param {string} email
654 	@returns {boolean} status
655 */
656 function echeck(str) {
657 
658 		var at="@"
659 		var dot="."
660 		var lat=str.indexOf(at)
661 		var lstr=str.length
662 		var ldot=str.indexOf(dot)
663 		
664 		if (str.indexOf(at)==-1) return false;
665 		if (str.indexOf(at)==-1 || str.indexOf(at)==0 || str.indexOf(at)==lstr) return false;
666 		if (str.indexOf(dot)==-1 || str.indexOf(dot)==0 || str.indexOf(dot)==lstr) return false;
667 		if (str.indexOf(at,(lat+1))!=-1) return false;
668 		if (str.substring(lat-1,lat)==dot || str.substring(lat+1,lat+2)==dot) return false;
669 		if (str.indexOf(dot,(lat+2))==-1) return false;
670 		if (str.indexOf(" ")!=-1) return false;
671 
672  		return true;
673 	}

687 	return true;
